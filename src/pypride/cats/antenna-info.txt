#   6)  49:55  -- Measurement type ME_COMP (complete) ME_INCM (incomplete)

#   7)  58:61  -- Reference temperature (degree C)  T0
#   8)  63:66  -- sin amplitude of annual temperature variations with respect 
#                 to the J2000.0 epoch (degree C) sin_T
#   9)  68:71  -- cos amplitude of annual temperature variations with respect 
#                 to the J2000.0 epoch (degree C) cos_T
#  10)  73:78  -- Reference pressure (hPa) h0
#  11)  81:85  -- Antenna diameter (m) ant_diam
#  12)  87:93  -- Height of foundation (m)  hf
#  13)  95:100 -- Depth of foundation (m) df
#  14) 103:109 -- Foundation thermal expansion coefficient (1/K) gamma_hf
#  15) 112:118 -- Length of the fixed axis (m) hp
#  16) 120:126 -- Fixed axis thermal expansion coefficient (1/K) gamma_hp
#  17) 129:135 -- Length of the axis offset (m) AO
#  18) 137:143 -- Axis offset thermal expansion coefficient (1/K) gamma_AO
#  19) 146:152 -- Distance from the movable axis to the antenna vertex (m)  hv
#  20) 154:160 -- Thermal expansion coefficient of the structure from

#  21) 163:169 -- Height of the sub-reflector/primary focus above the vertex (m)  hs
